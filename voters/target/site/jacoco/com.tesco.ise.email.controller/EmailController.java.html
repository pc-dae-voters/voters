package com.tesco.ise.email.controller;

import com.tesco.ise.email.dto.ContactAPIRequest;
import com.tesco.ise.email.dto.ContactApiResponse;
import com.tesco.ise.email.service.EmailService;
import com.tesco.ise.email.utils.EmailException;
import io.swagger.v3.oas.annotations.Operation;
import io.swagger.v3.oas.annotations.tags.Tag;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.http.HttpStatus;
import org.springframework.http.HttpStatusCode;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.*;

/** Default resource controller class with some sample endpoints. */
@RestController
@RequestMapping("")
@Slf4j
public class EmailController {

  @Value("${spring.application.name}")
  private String appName;

  @Autowired private EmailService emailService;

  /**
   * Send email.
   *
   * @return 202
   */
  @Operation(
      summary = "Send an email via the Contact API service.",
      description =
          "Pass in receipent and content data as well as templateId to send an email using Contact API service.")
  @PostMapping(value = "/send/{templateId}", produces = "application/json")
  @ResponseStatus(HttpStatus.ACCEPTED)
  @Tag(name = "Email")
  public ResponseEntity<String> sendEmail(
      @PathVariable String templateId, @RequestBody ContactAPIRequest emailData) {
    log.trace("calling with arguments, templateId: {}, emailData: {}", templateId, emailData);
    try {
      ContactApiResponse resp = emailService.sendEmail(templateId, emailData);
      return new ResponseEntity<>(resp.getBody(), resp.getStatus());
    } catch (EmailException e) {
      return new ResponseEntity<>(e.getMessage(), HttpStatusCode.valueOf(500));
    }
  }
}
