package com.tesco.ise.email.service.impl;

import com.tesco.ise.email.dto.ContactAPIRequest;
import com.tesco.ise.email.dto.ContactApiResponse;
import com.tesco.ise.email.feign.contactapiclient.ContactApiGateway;
import com.tesco.ise.email.service.EmailService;
import com.tesco.ise.email.utils.EmailException;
import feign.Response;
import feign.Util;
import java.nio.charset.Charset;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.http.HttpStatusCode;
import org.springframework.stereotype.Service;

@Service
@Slf4j
@RequiredArgsConstructor
public class EmailServiceImpl implements EmailService {

  private final ContactApiGateway contactApiGateway;

  private String getBody(Response resp) throws EmailException {
    try {
      return Util.toString(resp.body().asReader(Charset.defaultCharset()));
    } catch (Exception e) {
      throw new EmailException("Failed to deserialize error response", e);
    }
  }

  @Override
  public ContactApiResponse sendEmail(String templateId, ContactAPIRequest emailData)
      throws EmailException {
    log.trace("calling with arguments, templateId: {}, emailData: {}", templateId, emailData);
    try {
      Response resp = contactApiGateway.sendEmail(templateId, emailData);
      String body = getBody(resp);
      log.debug("Response code: {}", resp.status());
      log.debug("Body: {}", body);
      return new ContactApiResponse(HttpStatusCode.valueOf(resp.status()), body);
    } catch (Exception e) {
      throw new EmailException("failed to send email", e);
    }
  }
}
